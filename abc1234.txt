import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.Scanner;

public class Lab {

	public static void main(String[] args) throws IOException {
		Scanner input = new Scanner(System.in);
		System.out.print("Input word to count occurences of: ");
		String word = input.nextLine();
		System.out.println("Searching...");
		System.out.println("Occurences: " + countOccurences(word, "src/shakespeare.txt"));
		input.close();
		
		
	}
	
	static int countOccurences(String word, String filename) throws IOException {
		BufferedReader br = new BufferedReader(new FileReader(filename));
		int count = 0;
		try {
			String line = br.readLine();

			while (line != null) {
				String[] words = line.split(" ");
				for (int i = 0; i < words.length; i++) {
					if (words[i].toLowerCase().contains(word.toLowerCase())) {
						count++;
					}
				}
				line = br.readLine();
			}
		} finally {
			br.close();
		}
		
		return count;
	}
}
